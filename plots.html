data("rest_inspec")


scores_2017 = 
  rest_inspec %>% 
  separate(inspection_date, c("inspection_year", "inspection_month", "inspection_day"), sep = "-") %>% 
  filter(inspection_year == 2017 & score != "NA" & boro != "NA" & boro != "Missing") %>%
  distinct(camis, score, boro)
  
  
scores_2017 %>% plot_ly(y = ~score, color = ~boro, type = "box", colors = "Set2")
scores_plot = scores_2017 %>% 
  ggplot(aes(y = score, color = boro)) +
    geom_boxplot() +
    labs(
      x = "Boro of NYC",
      y = "Inspection Score",
      title = "Distribution of 2017 NYC Restaurant Inspections Scores by Boro"
    ) +
    viridis::scale_color_viridis(
      name = "Boro",
      discrete = TRUE
    ) +
    theme_bw() +
    theme(axis.text.x = element_blank(),
          axis.ticks.x = element_blank()
    )
